#include <iostream>
#include <set>
using namespace std;

int solution(int a[], int n) {
    int L = a[0], R = a[0];
    set<int> unique_elements;

    // Tìm L và R và lưu trữ các phần tử trong set
    for (int i = 0; i < n; i++) {
        L = min(L, a[i]);
        R = max(R, a[i]);
        unique_elements.insert(a[i]);
    }

    // Tính số lượng các số thiếu trong khoảng [L, R]
    int missing_count = 0;
    for (int i = L; i <= R; i++) {
        if (unique_elements.find(i) == unique_elements.end()) {
            missing_count++;
        }
    }

    return missing_count;
}

int main() {
    int n;
    cin >> n;
    
    int a[n];
    
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    cout << solution(a, n) << endl;
    
    return 0;
}
