int n,k;
int a[MAX];
void swap(int &a,int &b){
    int temp=a;
    a=b;
    b=temp;
}
int partition(int start, int end){
    int i_small=start-1;
    int pivot_val= a[end];
    for(int i=start;i<end;i++){
        if(a[i]<pivot_val){
            i_small++;
            swap(a[i],a[i_small]);
        }
    }
    swap(a[end],a[i_small+1]);
    return i_small+1;
}
void quick_sort(int start,int end){
    if(start>=end) return;
    int pivot= partition(start,end);
    quick_sort(start,pivot-1);
    quick_sort(pivot+1,end);
}

int Bai6_1(){
    quick_sort(0,n-1);
    int sum=0;
    for(int i=n-1;k>0;){
        sum+=a[i];
        k--;
        i--;
    }
    return sum;
}
