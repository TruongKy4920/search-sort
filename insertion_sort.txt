#define MAX 100
using namespace std;

std::ifstream input("input.txt");
std::ofstream output("output.txt");

int n, p;
int a[MAX];

void insert_sort(int x ,int j){
    
    for(;j>0;j--){
        if(x> a[j-1]) {
            break;
        }
        a[j]=a[j-1];      
    }
    a[j]=x;
}

void solution() {
    for(int i=0;i<n;i++){
         insert_sort(a[i],i);
    }
     for(int i=0;i<n;i++){
         output<<a[i]<<" ";
    }

}
