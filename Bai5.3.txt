int n,countsquare; 
int a[MAX];
int b[MAX];
bool checkSquare(int x){
    int cp= sqrt(x);
    return (pow(cp,2)==x);
}

void add_chinh_phuong(){
    countsquare =0;
    for(int i=0;i<n;i++){
        if(a[i]==0) continue;
        if(checkSquare(a[i])){
            b[countsquare+1]=a[i];
            countsquare++;
        } 
    }
}

bool binary_search(int s,int e,int x){
    while(s<=e){
        int mid=(s+e)/2;
        if(b[mid]==x) return 1;
        
        if(x > b[mid]) s=mid+1;
        else e=mid-1;
    }
    return 0;
}
int Bai5_3(){
    add_chinh_phuong();
     int res;
    for(int i=1;i<=countsquare; i++){
        cout<<i<<" ";
        if(binary_search(1,countsquare,i*i)){
            if(i==countsquare)  res= ++i;
            continue;
        }
        else return i*i;
    }
    return res*res;
}
