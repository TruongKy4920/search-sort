#define MAX 100

std::ifstream input("input.txt");
std::ofstream output("output.txt");
int n,x;
int a[MAX];
int binary_searh_recur(int s,int e){ 
        int mid = (s+e)/2;
        if(a[mid]==x) return a[mid];

        if(x> a[mid]){
            s=mid+1;
            return binary_searh_recur(s,e);
        } 
        else {
            e=mid-1;
            return binary_searh_recur(s,e);
        } 
 return -1;
}
void solution(){
    output<<binary_searh_recur(0,n);
}

      for(int i=0;i<n;i++){
        input>>a[i];
      }
