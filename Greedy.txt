#include <iostream>
#include <fstream>
#include <string>
#include <sstream>
#include <iomanip>
#include <math.h>
#include <algorithm>
#include <stack>
#define MAX 100
using namespace std;

std::ifstream input("input.txt");
std::ofstream output("output.txt");

struct Item{
  string name;
  float price, weight, unit_price;
  int quantity;
};
Item a[MAX];
int n;
float max_weight;
void swap(Item &a,Item &b){
  Item temp=a;
  a=b;
  b=temp;
}
void sort_unit_price(){
  for(int i=0;i<n-1;i++){
    for(int j=i+1;j<n;j++){
      if(a[i].unit_price< a[j].unit_price) swap(a[i],a[j]);
    }
  }
}
void print(){
  for(int i=0;i<n;i++){
    output<<a[i].name<<": "<<a[i].quantity<<"\n";
  }
}

void solution(){
  sort_unit_price(); //xep theo ti gia giam dan
  for(int i=0;i<n;i++){
    a[i].quantity=max_weight/a[i].weight;//lay khoi luong toi da bao tai chua duoc chia lay nguyen voi 
                                          //khoi luong cua tung item
    max_weight-=(a[i].quantity*a[i].weight);//update lai khoi luong bao tai sau khi them item
  }
  print();
}

int main(){
    std::string line;
    input>>n>>max_weight;
    //Buoc 1 tinh ti gia cua tung item va xep thang co ti gia lon
    //nhat o dau mang
   for(int i=0;i<n;i++){
    input>>a[i].name>>a[i].weight>>a[i].price;
    a[i].unit_price=a[i].price/a[i].weight;
    a[i].quantity=0; // voi moi item cho vao list thi cho so luong bang khong
   }
    solution();


  


    
      
      
      
}
