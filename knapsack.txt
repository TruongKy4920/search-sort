#include <iostream>
#include <fstream>
#include <string>
#include <sstream>
#include <iomanip>
#include <math.h>
#include <algorithm>
#include <stack>
#define MAX 100
using namespace std;

std::ifstream input("input.txt");
std::ofstream output("output.txt");
int n,sack_limit;

struct Item{
  string id;
  int price,weight;
};
int a[MAX][MAX];
Item items[MAX];

void solution(){
  for(int i=0;i<=n;i++)  a[i][0]=0;
  for(int j=0;j<=sack_limit;j++) a[0][j]=0;
  for(int i=1;i<=n;i++){
    for(int j=1;j<=sack_limit;j++){
      int not_select_price=a[i-1][j];
      int select_price=items[i].price+a[i-1][j-(items[i].weight)];
      if(items[i].weight<j){
          a[i][j]=max(not_select_price,select_price);
      }
      else a[i][j]=not_select_price;
    }
  }
  output<<a[n][sack_limit];
}

int main(){
    std::string line;
    input>>n>>sack_limit;
    for(int i=1;i<=n;i++){
      input>>items[i].id>>items[i].weight>>items[i].price;
    }
    solution();
    for(int i=0;i<=n;i++){
        for(int j=0;j<=sack_limit;j++){
          cout<<a[i][j]<<"    ";
      }
      cout<<"\n";
    }
  
  }
